import feedparser
def search_arxiv(query="quantum gravity", max_results=5):
    base_url = "http://export.arxiv.org/api/query?"
    search_query = f"search_query=all:{query}&start=0&max_results={max_results}"
    response = feedparser.parse(base_url + search_query)
    results = []
    for entry in response.entries:
        results.append({
            "title": entry.title,
            "summary": entry.summary,
            "url": entry.link,
            "authors": [author.name for author in entry.authors]
        })
    return results

import mimetypes
def detect_file_type(file_path):
    mime_type, _ = mimetypes.guess_type(file_path)
    if mime_type is None:
        ext = os.path.splitext(file_path)[-1]
        return ext.lower()
    return mime_type
def route_to_parser(file_path, file_type):
    if "pdf" in file_type:
        return parse_pdf(file_path)
    elif "word" in file_type or file_type.endswith(".docx"):
        return parse_docx(file_path)
    elif "json" in file_type:
        return parse_json(file_path)
    elif "csv" in file_type:
        return parse_csv(file_path)
    elif file_type in [".py", ".js", ".html"]:
        return parse_code(file_path)
    elif "image" in file_type:
        return parse_image(file_path)

from PyPDF2 import PdfReader
from docx import Document
import json
import csv
def parse_pdf(path):
    reader = PdfReader(path)
    return "\n".join([page.extract_text() for page in reader.pages])
def parse_docx(path):
    doc = Document(path)
    return "\n".join([p.text for p in doc.paragraphs])
def parse_json(path):
    with open(path) as f:
        return json.load(f)
def parse_csv(path):
    with open(path) as f:
        return list(csv.reader(f))
def parse_code(path):
    with open(path) as f:
        return f.read()

from transformers import AutoTokenizer
tokenizer = AutoTokenizer.from_pretrained("gpt2")  # or other model
def chunk_messages(messages, max_tokens=512):
    chunks = []
    current_chunk = ""
    for msg in messages:
        chunk_try = current_chunk + f"\n[{msg['role']}]: {msg['text']}"
        if len(tokenizer.encode(chunk_try)) > max_tokens:
            chunks.append(current_chunk.strip())
            current_chunk = f"[{msg['role']}]: {msg['text']}\n"
        else:
            current_chunk = chunk_try
    if current_chunk:
        chunks.append(current_chunk.strip())
    return chunks

import re
from pathlib import Path
def classify_and_save(messages, base_path="projects/ChatGPT_Import"):
    topics = {
        "books": [],
        "courses": [],
        "code": [],
        "science": [],
        "trusts": [],
        "misc": []
    }
    for i, msg in enumerate(messages):
        lower = msg.lower()
        if "chapter" in lower or "book title" in lower:
            topics["books"].append(msg)
        elif "module" in lower or "course" in lower:
            topics["courses"].append(msg)
        elif "def " in lower or "import " in lower:
            topics["code"].append(msg)
        elif "equation" in lower or "theory" in lower:
            topics["science"].append(msg)
        elif "trust" in lower or "LLC" in lower or "filing":
            topics["trusts"].append(msg)
        else:
            topics["misc"].append(msg)
    Path(base_path).mkdir(parents=True, exist_ok=True)
    for topic, entries in topics.items():
        with open(f"{base_path}/{topic}.txt", "w", encoding="utf-8") as f:
            f.write("\n\n---\n\n".join(entries))

import os
import json
def build_jsonl_chunks(source_txt, output_prefix, chunk_token_limit=250_000, tokenizer=None):
    with open(source_txt, 'r', encoding='utf-8') as f:
        data = f.read().split("\n\n---\n\n")
    os.makedirs("fine_tune_chunks", exist_ok=True)
    jsonl_chunk = []
    token_count = 0
    file_count = 1
    for item in data:
        prompt = "User: Analyze or continue this content:\n" + item.strip()
        completion = "Assistant: " + item.strip()
        tokens = len(tokenizer.encode(prompt + completion)) if tokenizer else len(prompt.split())
        if token_count + tokens > chunk_token_limit:
            with open(f"fine_tune_chunks/{output_prefix}_{file_count}.jsonl", "w", encoding="utf-8") as out:
                for line in jsonl_chunk:
                    out.write(json.dumps(line) + "\n")
            jsonl_chunk = []
            token_count = 0
            file_count += 1
        jsonl_chunk.append({"prompt": prompt, "completion": completion})
        token_count += tokens
    if jsonl_chunk:
        with open(f"fine_tune_chunks/{output_prefix}_{file_count}.jsonl", "w", encoding="utf-8") as out:
            for line in jsonl_chunk:
                out.write(json.dumps(line) + "\n")

import os
import json
def file_size_mb(path):
    return os.path.getsize(path) / (1024 * 1024)
def build_jsonl_chunks(input_txt_file, output_prefix, chunk_size_mb=500):
    with open(input_txt_file, 'r', encoding='utf-8') as f:
        entries = f.read().split("\n\n---\n\n")
    os.makedirs("fine_tune_chunks", exist_ok=True)
    chunk = []
    file_index = 1
    temp_path = f"fine_tune_chunks/{output_prefix}_{file_index}.jsonl"
    size_limit_bytes = chunk_size_mb * 1024 * 1024
    for entry in entries:
        prompt = "User: Continue or complete the following.\n"
        completion = entry.strip()
        item = {
            "prompt": prompt,
            "completion": completion
        }
        chunk.append(item)
        # Estimate size of current chunk
        with open(temp_path, "w", encoding="utf-8") as f:
            for c in chunk:
                f.write(json.dumps(c) + "\n")
        if file_size_mb(temp_path) >= chunk_size_mb:
            print(f"✅ Saved chunk: {temp_path}")
            file_index += 1
            temp_path = f"fine_tune_chunks/{output_prefix}_{file_index}.jsonl"
            chunk = []
    # Save final file if anything left
    if chunk:
        with open(temp_path, "w", encoding="utf-8") as f:
            for c in chunk:
                f.write(json.dumps(c) + "\n")
        print(f"✅ Final chunk saved: {temp_path}")

def chunk_jsonl(input_file, output_prefix, chunk_size=10000):
    with open(input_file, 'r', encoding='utf-8') as infile:
        lines = []
        part = 1
        for i, line in enumerate(infile, 1):
            lines.append(line)
            if i % chunk_size == 0:
                with open(f"{output_prefix}_part{part}.jsonl", 'w', encoding='utf-8') as out:
                    out.writelines(lines)
                lines = []
                part += 1
        if lines:
            with open(f"{output_prefix}_part{part}.jsonl", 'w', encoding='utf-8') as out:
                out.writelines(lines)

import json
import os
from datetime import datetime
MEMORY_FILE = "memory/apollo_forever_memory.jsonl"
def log_entry(project, chat_id, user_input, model_output, model_used="apollo", file_refs=None, tags=None):
    entry = {
        "timestamp": datetime.utcnow().isoformat(),
        "project": project,
        "chat_id": chat_id,
        "user_input": user_input,
        "model_output": model_output,
        "model_used": model_used,
        "file_refs": file_refs or [],
        "tags": tags or []
    }
    os.makedirs(os.path.dirname(MEMORY_FILE), exist_ok=True)
    with open(MEMORY_FILE, "a", encoding="utf-8") as f:
        f.write(json.dumps(entry) + "\n")
def load_memory(project=None, chat_id=None):
    if not os.path.exists(MEMORY_FILE):
        return []
    results = []
    with open(MEMORY_FILE, "r", encoding="utf-8") as f:
        for line in f:
            entry = json.loads(line)
            if project and entry.get("project") != project:
                continue
            if chat_id and entry.get("chat_id") != chat_id:
                continue
            results.append(entry)
    return results

from fastapi import FastAPI, Request
from pydantic import BaseModel
import requests
app = FastAPI()
class PromptRequest(BaseModel):
    prompt: str
    model: str = "mistral"
OLLAMA_ENDPOINT = "http://localhost:11434/api/generate"
def generate_with_ollama(prompt, model_name="mistral"):
    res = requests.post(
        OLLAMA_ENDPOINT,
        json={"model": model_name, "prompt": prompt, "stream": False}
    )
    return res.json()["response"]
@app.post("/ask")
async def ask_route(request: PromptRequest):
    output = generate_with_ollama(request.prompt, request.model)
    return {"model": request.model, "output": output}

import json
import os

# Load LoRA metadata
with open("metadata/adapter_metadata.json") as f:
    adapters = json.load(f)
def route_lora(prompt):
    prompt_vec = model.encode(prompt, convert_to_tensor=True)
    best_match = max(
        adapters, key=lambda l: util.cos_sim(prompt_vec, model.encode(l["intent_vector"], convert_to_tensor=True))
    )
    return best_match
